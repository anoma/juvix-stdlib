module Stdlib.Data.String.Ord;
  open import Stdlib.Data.String;
  open import Stdlib.Data.Bool;

  foreign c {
  bool cStrEq(char* l, char* r) {
    return strcmp(l, r) == 0;
  \}
  };

  axiom eqString : String → String → Bool;

  compile eqString {
    c ↦ "cStrEq";
  };

  infix 4 ==;
  == : String → String → Bool;
  == s1 s2 := eqString s1 s2;
end;
