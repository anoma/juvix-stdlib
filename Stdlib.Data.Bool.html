module Stdlib.Data.Bool;
  builtin bool type Bool :=
    | true : Bool
    | false : Bool;

  not : Bool → Bool;
  not true := false;
  not false := true;

  infixr 2 ||;
  || : Bool → Bool → Bool;
  || false a := a;
  || true _ := true;

  infixr 2 &&;
  && : Bool → Bool → Bool;
  && false _ := false;
  && true a := a;

  builtin bool-if if : {A : Type} → Bool → A → A → A;
  if true a _ := a;
  if false _ b := b;
end;
