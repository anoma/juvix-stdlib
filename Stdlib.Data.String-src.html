module Stdlib.Data.String;

import Stdlib.Data.String.Base open public;

import Stdlib.Trait.Eq as Eq;
open Eq using {Eq};

import Stdlib.Trait.Show as Show;
open Show using {Show};

import Stdlib.Data.String.Ord as String;

module StringTraits;
  
  Eq : Eq String;
  Eq := Eq.mkEq (String.==);
  
  Show : Show String;
  Show :=
    let
      go : String -> String;
      go s := "\"" ++str s ++str "\"";
    in Show.mkShow go;

end;
