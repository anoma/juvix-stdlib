head {A} : List A ‚Üí A
  | (x :: _) := x
  | nil := fail "head: empty list";

module ListTraits;
  Eq {A} : Eq.Eq A -> Eq.Eq (List A)
    | (Eq.mkEq eq-a) :=
      let
        go : List A -> List A -> Bool
          | nil nil := true
          | nil _ := false
          | _ nil := false
          | (x :: xs) (y :: ys) := if (eq-a x y) (go xs ys) false;
      in Eq.mkEq go;
  
  Ord {A} : Ord.Ord A -> Ord.Ord (List A)
    | (Ord.mkOrd cmp-a) :=
      let
        go : List A -> List A -> Ord.Ordering
          | nil nil := Ord.EQ
          | nil _ := Ord.LT
          | _ nil := Ord.GT
          | (x :: xs) (y :: ys) :=
            case cmp-a x y
              | Ord.LT := Ord.LT
              | Ord.GT := Ord.GT
              | Ord.EQ := go xs ys;
      in Ord.mkOrd go;
  
  Show {A} : Show.Show A -> Show.Show (List A)
    | (Show.mkShow to-str) :=
      let
        go : List A -> String
          | nil := "nil"
          | (x :: xs) := to-str x ++str " :: " ++str go xs;
      in Show.mkShow
        Œª {
          | nil := "nil"
          | s := "(" ++str go s ++str ")"
        };
end;
