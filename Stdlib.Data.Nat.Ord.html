module Stdlib.Data.Nat.Ord;
  open import Stdlib.Data.Nat;
  open import Stdlib.Data.Ord;

  compare : ℕ → ℕ → Ordering;
  compare zero zero ≔ EQ;
  compare zero _ ≔ LT;
  compare (suc _) zero ≔ GT;
  compare (suc a) (suc b) ≔ compare a b;
end;
