module Stdlib.Data.Maybe;
  inductive Maybe (A : Type) {
    nothing : Maybe A;
    just : A → Maybe A;
  };

  fromMaybe : {A : Type} → A → Maybe A → A;
  fromMaybe a nothing ≔ a;
  fromMaybe _ (just a) ≔ a;

  maybe : {A : Type} → {B : Type} → B → (A → B) → Maybe A → B;
  maybe b _ nothing ≔ b;
  maybe _ f (just a) ≔ f a;
end;
