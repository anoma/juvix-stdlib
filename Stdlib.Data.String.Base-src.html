--- Primitive representation of a sequence of characters.
builtin string
axiom String : Type;

infixr 5 ++str;
--- Concatenation of two ;String;s.
builtin string-concat
axiom ++str : String -> String -> String;

--- Concatenates a ;List; of ;String;s.
concatStr : List String -> String;
concatStr := foldl (++str) "";
